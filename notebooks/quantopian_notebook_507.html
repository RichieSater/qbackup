def pd_frama(c, per):
    c = c.copy()
    window = per * 2

    hh = c.rolling(per).max()
    ll = c.rolling(per).min()

    n1 = (hh - ll) / per
    n2 = n1.shift(per)

    hh2 = c.rolling(window).max()
    ll2 = c.rolling(window).min()
    n3 = (hh2 - ll2) / window

    D = (np.log(n1 + n2) - np.log(n3)) / np.log(2)
    alp = np.exp(-4.6 * (D - 1))
    alp = np.clip(alp, .01, 1).values

    filt = c.values
    for i, x in enumerate(alp):
        cl = c.values[i]
        if i < window:
            continue
        filt[i] = cl * x + (1 - x) * filt[i - 1]

    return filt
