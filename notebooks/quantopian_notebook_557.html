def make_pipeline():
    sentiment_factor = sentiment.sentiment_signal.latest
    
    universe = (Q1500US() & sentiment_factor.notnull())
    
    pipe = Pipeline(columns={'sentiment': sentiment_factor,
                            'longs': (sentiment_factor>=4),
                            'shorts': (sentiment_factor<=-2)},
                   screen=universe)
    return pipe
